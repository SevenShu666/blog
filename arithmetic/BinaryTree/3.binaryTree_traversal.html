# 3.二叉树的迭代遍历

# 前序遍历

力扣题目链接 (opens new window)

遍历顺序：中左右

var preorderTraversal = function (root) {
  if (!root) return [];
  const stack = [root];
  const res = [];

  while (stack.length) {
    const node = stack.pop();
    res.push(node.val);
    if (node.right) {
      stack.push(node.right);
    }
    if (node.left) {
      stack.push(node.left);
    }
  }

  return res;
};

# 中序遍历

力扣题目链接 (opens new window)

遍历顺序：左中右

var inorderTraversal = function (root) {
  const res = [];
  const stack = [];
  let cur = root;

  while (cur || stack.length) {
    while (cur) {
      stack.push(cur);
      cur = cur.left;
    }

    const node = stack.pop();
    res.push(node.val);
    cur = node.right;
  }

  return res;
};

# 后续遍历

力扣题目链接 (opens new window)

遍历顺序：左右中

var postorderTraversal = function (root) {
  if (!root) return [];
  const res = [];
  const stack = [root];

  while (stack.length) {
    const node = stack.pop();
    res.unshift(node.val);
    if (node.left) {
      stack.push(node.left);
    }
    if (node.right) {
      stack.push(node.right);
    }
  }

  return res;
};
